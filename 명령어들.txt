# 파이썬 폴더 터미널에서 해야할 것들

python -m venv venv -> 가상환경 만들기
	
.\venv\Scripts\activate -> 가상환경 실행, 앞에 venv가 뜨면 실행중인거

python -m pip install --upgrade pip	-> 필요 라이브러리들 install 하는거
python -m pip install pandas joblib fastapi uvicorn scikit-learn xgboost

python main.py -> main.py 실행, FastAPI 실행됨

# Front 터미널 명령어
cd FRONT
npm install
npm install @ckeditor/ckeditor5-react @ckeditor/ckeditor5-build-classic
npm install recharts
npm install react-kakao-maps-sdk
npm start
